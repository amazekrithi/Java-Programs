public class numbersort {
	public static void main(String[] arg){
		int a[]=new int[100];
		int b,k=1;
		Scanner jaya =new Scanner(System.in);
		b=jaya.nextInt();
		for(int i=0;i<b;i++){
			a[i]=jaya.nextInt();
		}
		for(int i=0;i<b;i++){
			for(int j=i+1;j<b;j++){
				if(a[i]==a[j]){
					k++;
				}
			}
			if(k>1){System.out.println(a[i]+" "+k);return;}
		}
		System.out.println("NODUPLICATEELEMENT");
	}
}
